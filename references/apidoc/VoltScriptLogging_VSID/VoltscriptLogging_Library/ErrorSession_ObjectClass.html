      • getAndRemoveLastError

        Public Function getAndRemoveLastError() As ErrorType
        Gets and removes the last error added to the stack
         
        Code Sample
        ' This code writes logs for all errors previously captured and clears the stack
        Dim errorObj as ErrorType
        Set errorObj = getErrorSession().getAndRemoveLastError()
        While (Not errorObj Is Nothing) Then
            Call getLogSession().createLogEntry(LOG_WARNING, errorObj.getLogMessage(), errorObj.stackTrace, "")
            Set errorObj = getErrorSession().getAndRemoveLastError()
        Wend
